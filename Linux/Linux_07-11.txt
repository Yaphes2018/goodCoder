#!/bin/bash
echo "Please input your emailaddress:"

read email
str=`echo $email | egrep "^[a-zA-Z0-9_-\.\+]+@[a-zA-Z_-\.]+(\.[a-zA-Z0-9_-]+)+$"|wc -l`

if [ $str == 0 ];then

       echo "validation is wroing"

else

       echo "Your email address is $email"

fi

//-Q4：监控CPU使用，输出日期时间，CPU load，CPU idle。
#!/bin/bash
#以年月日 时分秒的形式输出系统时间
echo $(date +%F%n%T)
#输出cpu load
uptime

#输出 cpu_idle
top -b -n 1 | grep Cpu | awk '{print $1,$5}' | cut -f 1 -d "."
